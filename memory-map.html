        function lngLatToVector3(lng, lat, r = radius) {

            const phi = (90 - lat) * Math.PI / 180;

            return new THREE.Vector3(r * Math.sin(phi) * Math.cos(theta), r * Math.cos(phi), r * Math.sin(phi) * Math.sin(theta));
